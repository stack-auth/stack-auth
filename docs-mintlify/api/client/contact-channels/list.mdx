---
title: "List Contact Channels"
description: "Retrieves a list of all contact channels for a user."
api: "GET /api/v1/contact-channels"
---

## Request

<ParamField header="x-stack-publishable-client-key" type="string" required>
  The publishable client key for authentication.
</ParamField>

<ParamField header="x-stack-access-token" type="string" required>
  The access token for the authenticated user.
</ParamField>

<ParamField header="x-stack-project-id" type="string" required>
  The project ID.
</ParamField>

<ParamField query="user_id" type="string" required>
  The identifier of the user, or `me` to reference the currently authenticated user.
</ParamField>

<ParamField query="contact_channel_id" type="string">
  Contact channel identifier to filter by.
</ParamField>

## Response

<ResponseField name="items" type="array">
  A list of contact channels.
  <Expandable>
    <ResponseField name="id" type="string">
      The unique identifier of the contact channel.
    </ResponseField>
    <ResponseField name="user_id" type="string">
      The ID of the user who owns this contact channel.
    </ResponseField>
    <ResponseField name="type" type="string">
      The type of the contact channel (e.g., `email`).
    </ResponseField>
    <ResponseField name="value" type="string">
      The value of the contact channel (e.g., email address).
    </ResponseField>
    <ResponseField name="is_verified" type="boolean">
      Whether the contact channel has been verified.
    </ResponseField>
    <ResponseField name="is_primary" type="boolean">
      Whether this is the primary contact channel for the user.
    </ResponseField>
    <ResponseField name="used_for_auth" type="boolean">
      Whether this contact channel is used for authentication.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of when the contact channel was created.
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp of when the contact channel was last updated.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information.
  <Expandable>
    <ResponseField name="next_cursor" type="string | null">
      Cursor for the next page of results.
    </ResponseField>
  </Expandable>
</ResponseField>
