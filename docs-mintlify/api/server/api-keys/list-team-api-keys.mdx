---
title: "List Team API Keys"
description: "List all team API keys for the project with their metadata and status."
api: "GET /api/v1/team-api-keys"
---

## Request

<ParamField header="x-stack-secret-server-key" type="string" required>
  The secret server key for authentication.
</ParamField>

<ParamField header="x-stack-project-id" type="string" required>
  The project ID.
</ParamField>

<ParamField query="team_id" type="string" required>
  The identifier of the team whose API keys should be listed.
</ParamField>

## Response

<ResponseField name="items" type="array">
  A list of team API keys.
  <Expandable>
    <ResponseField name="id" type="string">
      The unique identifier of the API key.
    </ResponseField>
    <ResponseField name="description" type="string">
      A description of the API key.
    </ResponseField>
    <ResponseField name="team_id" type="string">
      The identifier of the team this API key belongs to.
    </ResponseField>
    <ResponseField name="expires_at" type="string">
      ISO 8601 timestamp of when the API key expires, or `null` if it does not expire.
    </ResponseField>
    <ResponseField name="manually_revoked_at" type="string">
      ISO 8601 timestamp of when the API key was manually revoked, or `null` if it has not been revoked.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of when the API key was created.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information.
  <Expandable>
    <ResponseField name="next_cursor" type="string | null">
      Cursor for the next page of results.
    </ResponseField>
  </Expandable>
</ResponseField>

<CodeGroup>
```bash cURL
curl -X GET 'https://api.stack-auth.com/api/v1/team-api-keys?team_id=TEAM_ID' \
  -H 'x-stack-secret-server-key: YOUR_SECRET_SERVER_KEY' \
  -H 'x-stack-project-id: YOUR_PROJECT_ID'
```
