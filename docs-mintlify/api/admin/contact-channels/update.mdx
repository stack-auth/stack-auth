---
title: "Update a Contact Channel"
description: "Modifies an existing contact channel by applying only the values included in the request payload, leaving other attributes unchanged."
api: "PATCH /api/v1/contact-channels/{user_id}/{contact_channel_id}"
---

## Request

<ParamField header="x-stack-secret-server-key" type="string" required>
  The secret server key for authentication.
</ParamField>

<ParamField header="x-stack-admin-access-token" type="string">
  The admin access token.
</ParamField>

<ParamField header="x-stack-project-id" type="string" required>
  The project ID.
</ParamField>

<ParamField path="user_id" type="string" required>
  The identifier for the user who owns the contact channel, or `me` for the currently authenticated user.
</ParamField>

<ParamField path="contact_channel_id" type="string" required>
  The identifier for the contact channel being modified.
</ParamField>

<ParamField body="value" type="string">
  The new value for the contact channel (e.g., a new email address).
</ParamField>

<ParamField body="is_primary" type="boolean">
  Whether this should be the primary contact channel for the user.
</ParamField>

<ParamField body="used_for_auth" type="boolean">
  Whether this contact channel should be used for authentication.
</ParamField>

<ParamField body="is_verified" type="boolean">
  Whether the contact channel is verified.
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the contact channel.
</ResponseField>

<ResponseField name="user_id" type="string">
  The ID of the user who owns this contact channel.
</ResponseField>

<ResponseField name="type" type="string">
  The type of the contact channel.
</ResponseField>

<ResponseField name="value" type="string">
  The value of the contact channel.
</ResponseField>

<ResponseField name="is_verified" type="boolean">
  Whether the contact channel has been verified.
</ResponseField>

<ResponseField name="is_primary" type="boolean">
  Whether this is the primary contact channel for the user.
</ResponseField>

<ResponseField name="used_for_auth" type="boolean">
  Whether this contact channel is used for authentication.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the contact channel was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the contact channel was last updated.
</ResponseField>

<CodeGroup>
```bash cURL
curl -X PATCH 'https://api.stack-auth.com/api/v1/contact-channels/usr_1234567890/cc_1234567890' \
  -H 'Content-Type: application/json' \
  -H 'x-stack-secret-server-key: YOUR_SECRET_SERVER_KEY' \
  -H 'x-stack-project-id: YOUR_PROJECT_ID' \
  -d '{
    "is_primary": true,
    "is_verified": true
  }'
```
